confirm() {
    local prompt="$1"
    local default="${2:-n}"
    
    if [[ "$default" == "y" ]]; then
        read -p "$prompt [Y/n]: " -n 1 -r
        echo
        if [[ -z "$REPLY" ]]; then
            return 0  # 默认选择Y
        else
            [[ $REPLY =~ ^[Yy]$ ]] && return 0
            return 1
        fi
    else
        read -p "$prompt [y/N]: " -n 1 -r
        echo
        if [[ -z "$REPLY" ]]; then
            return 1  # 默认选择N
        else
            [[ $REPLY =~ ^[Yy]$ ]] && return 0
            return 1
        fi
    fi
}

- **修复后改进**:
  - 检查前导零（如 01, 001 等）
  - 验证每个部分的数值范围
  - 防止格式错误的IP地址通过验证
  - 提高了安全性

**修复后的逻辑**:
```bash
validate_ipv4() {
    local ip="$1"
    
    # 检查基本格式
    if [[ ! $ip =~ ^[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}$ ]]; then
        return 1
    fi
    
    # 检查是否有前导零
    if [[ $ip =~ 0[0-9] ]]; then
        return 1
    fi
    
    # 检查每个部分的数值范围
    local IFS='.'
    local -a ip_parts=($ip)
    for part in "${ip_parts[@]}"; do
        # 检查是否为空或超过255
        if [[ -z "$part" || $part -gt 255 ]]; then
            return 1
        fi
        # 检查是否有前导零（除了单独的0）
        if [[ ${#part} -gt 1 && ${part:0:1} == "0" ]]; then
            return 1
        fi
    done
    
    return 0
}
```
